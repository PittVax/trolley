grouped = df.reset_index().groupby(['R0','primary_cases_timing','vaccination'])
for name, group in grouped:
    pivoted = group[['month', 'variable', 'monthly_clinical_attack_rate']].pivot(index='month',columns='variable',values='monthly_clinical_attack_rate')
    fig, ax = plt.subplots(1,1) 
    ax.get_xaxis().set_visible(False)
    pivoted.plot(table=np.round(pivoted.T, 4), ax=ax, title=name)
